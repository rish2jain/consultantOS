from pytrends.request import TrendReq
import pandas as pd

class GoogleTrendsTool:
    def __init__(self):
        self.pytrend = TrendReq(hl='en-US', tz=360)

    def get_interest_over_time(self, keywords: list[str], timeframe='today 12-m'):
        self.pytrend.build_payload(keywords, timeframe=timeframe)
        data = self.pytrend.interest_over_time()
        return data.to_dict()

    def get_interest_by_region(self, keywords: list[str]):
        self.pytrend.build_payload(keywords)
        data = self.pytrend.interest_by_region()
        return data.to_dict()

    def get_related_queries(self, keyword: str):
        self.pytrend.build_payload([keyword])
        data = self.pytrend.related_queries()
        return {
            'top': data[keyword]['top'].to_dict(),
            'rising': data[keyword]['rising'].to_dict()
        }
